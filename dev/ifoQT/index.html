Linear Quantum System - QuantumToolbox

using QuantumToolbox
N = 4
a = destroy(N) ⊗ qeye(N)
b = qeye(N) ⊗ destroy(N)

H1 = QobjEvo(b'*b, (p,t)->p[1])
H2 = QobjEvo(a'*b+a*b', (p,t)->p[2])

H = H1+H2

L = [QobjEvo(a, (p,t)->p[3])]
tlist = LinRange(0, 100, 1000)
p0 = [0.01,500,10]
ψ0= basis(N,0) ⊗ basis(N,0)

eops = [a, a'*a,b,b'*b]

sol = mesolve(H, ψ0, tlist, L;params = p0,e_ops=eops)
